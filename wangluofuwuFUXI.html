网络复习综合

网络服务综合

  • 创建网络
neutron net-create 20423-net
  • 创建子网
neutron subnet-create --name 20423-subnet --allocation-pool start=192.23.10.1,end=192.23.10.50 --disable-dhcp --gateway 192.23.10.254 20423-net 192.23.10.0/24
  • 修改子网
neutron subnet-update --gateway 192.23.20.254 --enable-dhcp --allocation-pool start=192.23.20.1,end=192.23.20.200 --name net20423-subnet 20423-net
  • 创建安全组
nova secgroup-create 20423-group '班级安全'
  • 给安全组加规则
nova secgroup-add-rule 20423-group ICMP -1 -1 0.0.0.0/0
nova secgroup-add-rule 20423-group TCP 65535 0.0.0.0/0
nova secgroup-add-rule 20423-group UDP 1 65535 0.0.0.0/0
  • 新建防火墙规则
neutron firewall-rule-create --name class23 --protocol tcp --source-port 1 --destination-port 65535 --action allow
neutron firewall-rule-create --name class23 --protocol udp --source-port 1 --destination-port 65535 --action allow
neutron firewall-rule-create --name class23 --protocol icmp --action reject
  • 修改规则
neutron firewall-rule-update --action allow <防火墙规则>
  • 把这个规则加进策略
neutron firewall-policy-create net23 --firewall-rules <防火墙规则>
  • 创建防火墙
neutron firewall-create --name net23 --router router <防火墙策略>
  • 创建命名空间
ip netns add nslsc
  • 创建网桥
ovs-vsctl add-br 23br
  • 创建内部通信
ovs-vsctl add-port 23br port23 -- set interface port23 type=internal
ip link set port23 netns ns23
ip netns exec ns23 ip addr add 172.114.23.1/24 dev port23
ip netns exec ns23 ip link set port23 up
ip netns exec ns23 ip link set lo up
  • 检测是否通信?
ip netns exec ns23 ping <ip> -c 4
